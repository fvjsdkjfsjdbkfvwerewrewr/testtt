        function updateProgressDOM() {

            const stats = JSON.parse(localStorage.getItem('networks_userStats'));

            const totalQuestions = questionBank.length;

            const solvedCount = Object.keys(stats.solved).length;

            

            document.getElementById('stat-solved').innerText = `${solvedCount} / ${totalQuestions}`;

            

            const totalAttempts = stats.correctCount + stats.wrongCount;

            const accuracy = totalAttempts > 0 ? Math.round((stats.correctCount / totalAttempts) * 100) : 0;

            document.getElementById('stat-accuracy').innerText = `${accuracy}%`;


            const percent = Math.round((solvedCount / totalQuestions) * 100);

            document.getElementById('stat-progress-bar').style.width = `${percent}%`;

        }


        function toggleStar(id) {

            let stats = JSON.parse(localStorage.getItem('networks_userStats'));

            if (stats.stars.includes(id)) {

                stats.stars = stats.stars.filter(starId => starId !== id);

                showToast("Pytanie usunięte z ulubionych");

            } else {

                stats.stars.push(id);

                showToast("Pytanie dodane do ulubionych");

            }

            localStorage.setItem('networks_userStats', JSON.stringify(stats));

            loadStarredQuestions();

            

            // Re-render current question views if active

            if (!document.getElementById('sec-study').classList.contains('hidden')) {

                loadStudyQuestion();

            }

        }

        function updateQuestionHistory(id, isCorrect) {

            } else {

        function startExam() {

            const countVal = document.getElementById('exam-count').value;

            const timeVal = document.getElementById('exam-time').value;


            // Generate customized array of randomized questions

            let shuffled = [...questionBank].sort(() => 0.5 - Math.random());

            let qCount = countVal === 'all' ? questionBank.length : parseInt(countVal);

            

            examState.questions = shuffled.slice(0, qCount);

            examState.currentIndex = 0;

            examState.answers = {};

            examState.active = true;


            // Handle timer configuration

            if (timeVal === 'no-limit') {

                examState.timeLimit = 0;

                document.getElementById('exam-timer-wrapper').classList.add('hidden');

            } else {

                examState.timeLimit = parseInt(timeVal) * 60;

                examState.timeRemaining = examState.timeLimit;

                document.getElementById('exam-timer-wrapper').classList.remove('hidden');

                updateExamTimerDisplay();

                

                // Clear any existing timer

                if (examState.timer) clearInterval(examState.timer);

                examState.timer = setInterval(() => {

                    examState.timeRemaining--;

                    updateExamTimerDisplay();

                    if (examState.timeRemaining <= 0) {

                        clearInterval(examState.timer);

                        finishExam(true);

                    }

                }, 1000);

            }


            document.getElementById('exam-setup').classList.add('hidden');

            document.getElementById('exam-active').classList.remove('hidden');

            loadExamQuestion();

        }

        }

        function startExam() {

            const countVal = document.getElementById('exam-count').value;

            const timeVal = document.getElementById('exam-time').value;


            // Generate customized array of randomized questions

            let shuffled = [...questionBank].sort(() => 0.5 - Math.random());

            let qCount = countVal === 'all' ? questionBank.length : parseInt(countVal);

            

            examState.questions = shuffled.slice(0, qCount);

            examState.currentIndex = 0;

            examState.answers = {};

            examState.active = true;


            // Handle timer configuration

            if (timeVal === 'no-limit') {

                examState.timeLimit = 0;

                document.getElementById('exam-timer-wrapper').classList.add('hidden');

            } else {

                examState.timeLimit = parseInt(timeVal) * 60;

                examState.timeRemaining = examState.timeLimit;

                document.getElementById('exam-timer-wrapper').classList.remove('hidden');

                updateExamTimerDisplay();

                

                // Clear any existing timer

                if (examState.timer) clearInterval(examState.timer);

                examState.timer = setInterval(() => {

                    examState.timeRemaining--;

                    updateExamTimerDisplay();

                    if (examState.timeRemaining <= 0) {

                        clearInterval(examState.timer);

                        finishExam(true);

                    }

                }, 1000);

            }


            document.getElementById('exam-setup').classList.add('hidden');

            document.getElementById('exam-active').classList.remove('hidden');

            loadExamQuestion();

        }

            document.getElementById('exam-timer').innerText = `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;

        }
